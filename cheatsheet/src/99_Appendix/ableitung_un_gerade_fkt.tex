\subsection{Ableitung Un-/Gerader Funktionen}
\begin{itemize}
    \item $g(x)$ sei \textbf{gerade}
        $$
            g'(0)=g^{(3)}(0)=g^{(5)}(0)=...=0
        $$
    \item $f(x)$ sei \textbf{ungerade}
        $$
            f''(0)=f^{(4)}(0)=f^{(6)}(0)=...=0
        $$
\end{itemize}
