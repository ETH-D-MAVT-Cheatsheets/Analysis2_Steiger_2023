\subsection{Komplex Konjugierte}
    \begin{align*}
        \text{Komplexe Zahl: } z = x - iy\\
        \text{Komplex konjugierte Zahl: } \bar{z} = x -iy\\
    \end{align*}
    \vspace{-2em}
    \[\begin{array}{c | c}
        \frac{z_1}{z_2} = \frac{z_1 \cdot \bar{z_2}}{z_2 \cdot \bar{z_2}} & \frac{1}{z^2 - 1} = \frac{\bar{z}^2 -1}{|z^2 - 1|}
    \end{array}\]
